	name = PTF_national_support_text
	text = {
		trigger = {
			original_tag = PTF
			check_variable = {

				value = 0.4
				compare = less_than
			}
		}
		localization_key = PTF_national_support_bad_loc

		trigger = {
			original_tag = PTF
			check_variable = {

				value = 0.4
				compare = greater_than_or_equals
			}
			check_variable = {

			}
