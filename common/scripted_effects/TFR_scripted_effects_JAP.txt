JAP_join_PEC = {
	if = {
		limit = {
			NOT = {
				has_country_flag = JAP_PEC_member_flag
			}
		}
		set_country_flag = JAP_PEC_member_flag
		if = {
			limit = {
				JAP = {
					has_completed_focus = JAP_promote_yen
				}
			}
			add_ideas = {
				JAP_PEC_member_2
			}
		}
		else = {
			add_ideas = {
				JAP_PEC_member
			}
		}
		remove_ideas = {
			ASEAN_member
			arab_league_member
		}
		JAP = {
			add_to_variable = {
				JAP_PEC_number_of_members = 1
			}
			add_to_variable = {
				JAP_PEC_consumer_goods_factor = -0.002
			}
			add_to_variable = {
				JAP_PEC_lend_lease_tension = -0.01
			}
			add_to_variable = {
				JAP_PEC_poverty_development_monthly = 0.001
			}
			add_to_variable = {
				JAP_PEC_industrial_development_monthly = 0.001
			}
			add_to_variable = {
				JAP_PEC_global_building_slots_factor = 0.005
			}
			add_to_variable = {
				JAP_PEC_production_speed_buildings_factor = 0.005
			}
		}
		add_to_array = {
			JAP.JAP_PEC_members = THIS
		}
		add_opinion_modifier = {
			target = JAP
			modifier = friendship_of_peoples
		}
		JAP = {
			add_opinion_modifier = {
				target = ROOT
				modifier = friendship_of_peoples
			}
		}
	}
}
