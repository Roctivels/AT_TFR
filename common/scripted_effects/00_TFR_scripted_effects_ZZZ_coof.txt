fourth_plague_pandemic_effect_initial_outbreak = {
	random_list = {
		1 = {
			random_neighbor_state  = {
				log = "Plague SPREADS to [THIS.GetName] from [PREV.GetName]"
				
				limit = {
					NOT = { is_in_array = { global.plague_affected_states = THIS.id } }
				}
				set_variable = { plague_state_modifier = -0.01 }
				add_dynamic_modifier = { modifier = plague_state_modifier }
				add_to_array = { array = global.plague_affected_states value = THIS.id }
			}
			
			add_to_variable = { global.plague_start_countdown = 1 }
			
			#Agriculture
			#modifier = { add = 0 controller = { has_idea = highest_farming } }
			modifier = { add = 1 controller = { has_idea = higher_farming } }
			modifier = { add = 3 controller = { has_idea = high_farming } }
			modifier = { add = 6 controller = { has_idea = medium_farming } }
			modifier = { add = 9 controller = { has_idea = low_farming } }
			modifier = { add = 12 controller = { has_idea = lower_farming } }
			
			modifier = { factor = 0 all_neighbor_state = { is_in_array = { global.plague_affected_states = THIS.id } } }
		}
		
		1 = {
			log = "Plague WORSENS to [THIS.GetName] from [PREV.GetName]"
			add_to_variable = { global.plague_start_countdown = 1 }
			
			#Agriculture
			#modifier = { add = 0 controller = { has_idea = highest_farming } }
			modifier = { add = 1 controller = { has_idea = higher_farming } }
			modifier = { add = 3 controller = { has_idea = high_farming } }
			modifier = { add = 6 controller = { has_idea = medium_farming } }
			modifier = { add = 9 controller = { has_idea = low_farming } }
			modifier = { add = 12 controller = { has_idea = lower_farming } }
			
			modifier = { factor = 0 all_neighbor_state = { is_in_array = { global.plague_affected_states = THIS.id } } }
		}
		
		5 = {
			log = "Plague IDLES at [THIS.GetName]"
		}
	}
}
