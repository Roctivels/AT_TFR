PTF_add_national_support = {
	custom_effect_tooltip = PTF_add_national_support_tooltip
	add_to_variable = {
		var = PTF_national_support_var
		value = PTF_national_support_var_temp
	}
	clamp_variable = {
		var = PTF_national_support_var
		min = 0
		max = 1
	}
	hidden_effect = {
		remove_ideas = PTF_lower_national_support_idea
		remove_ideas = PTF_low_national_support_idea
		remove_ideas = PTF_medium_national_support_idea
		remove_ideas = PTF_high_national_support_idea
		remove_ideas = PTF_higher_national_support_idea
		if = {
			limit = {
				check_variable = {
					var = PTF_national_support_var
					value = 0.2
					compare = less_than
				}
			}
			add_ideas = PTF_lower_national_support_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = PTF_national_support_var
					value = 0.2
					compare = greater_than_or_equals
				}
				check_variable = {
					var = PTF_national_support_var
					value = 0.4
					compare = less_than
				}
			}
			add_ideas = PTF_low_national_support_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = PTF_national_support_var
					value = 0.4
					compare = greater_than_or_equals
				}
				check_variable = {
					var = PTF_national_support_var
					value = 0.6
					compare = less_than
				}
			}
			add_ideas = PTF_medium_national_support_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = PTF_national_support_var
					value = 0.6
					compare = greater_than_or_equals
				}
				check_variable = {
					var = PTF_national_support_var
					value = 0.8
					compare = less_than
				}
			}
			add_ideas = PTF_high_national_support_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = PTF_national_support_var
					value = 0.8
					compare = greater_than_or_equals
				}
			}
			add_ideas = PTF_higher_national_support_idea
		}
	}
}
