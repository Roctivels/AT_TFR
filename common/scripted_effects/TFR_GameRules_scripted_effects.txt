	remove_ideas = {
		Custom_TFR_war_buff
		Custom_TFR_def_buff
		Custom_TFR_Crazy_war_buff 
		Custom_TFR_Crazy_def_buff 
	}
	delete_unit_template_and_units = {
		division_template = "Mushroom Army"
		disband = no
	}
	if = {
		limit = {
			has_template = "Mushroom Army 2"
		}
		delete_unit_template_and_units = {
			division_template = "Mushroom Army 2"
			disband = no
		}
	}
