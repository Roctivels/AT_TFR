USB_add_corporate_influence = {
	if = {
		limit = {
			NOT = {
				has_country_flag = USB_corporate_influence_disabled_flag
			}
		}
		custom_effect_tooltip = USB_add_corporate_influence_tooltip
		add_to_variable = {
			var = USB_corporate_influence_var
			value = USB_corporate_influence_var_temp
		}
		clamp_variable = {
			var = USB_corporate_influence_var
			min = 0
			max = 1
		}
		hidden_effect = {
			remove_ideas = USB_lower_corporate_influence_idea
			remove_ideas = USB_low_corporate_influence_idea
			remove_ideas = USB_medium_corporate_influence_idea
			remove_ideas = USB_high_corporate_influence_idea
			remove_ideas = USB_higher_corporate_influence_idea
			if = {
				limit = {
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.2
						compare = less_than
					}
				}
				add_ideas = USB_lower_corporate_influence_idea
			}
			else_if = {
				limit = {
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.2
						compare = greater_than_or_equals
					}
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.4
						compare = less_than
					}
				}
				add_ideas = USB_low_corporate_influence_idea
			}
			else_if = {
				limit = {
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.4
						compare = greater_than_or_equals
					}
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.6
						compare = less_than
					}
				}
				add_ideas = USB_medium_corporate_influence_idea
			}
			else_if = {
				limit = {
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.6
						compare = greater_than_or_equals
					}
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.8
						compare = less_than
					}
				}
				add_ideas = USB_high_corporate_influence_idea
			}
			else_if = {
				limit = {
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.8
						compare = greater_than_or_equals
					}
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.95
						compare = less_than
					}
				}
				add_ideas = USB_higher_corporate_influence_idea
			}
			else_if = {
				limit = {
					NOT = {
						has_country_flag = USB_reunified_flag
					}
					check_variable = {
						var = USB_corporate_influence_var
						value = 0.95
						compare = greater_than
					}
				}
				country_event = {
					id = usb.17
					hours = 6
				}
			}
		}
	}
