NSM_add_to_coalition = {
	NSM = {
		add_to_faction = FROM
		custom_effect_tooltip = USA_add_to_coalition_tooltip
		hidden_effect = {
			every_enemy_country = {
				declare_war_on = {
					target = FROM
					type = civil_war
				}
			}
		}
	}
	hidden_effect = {
		every_enemy_country = {
			every_other_country = {
				limit = {
					is_in_faction_with = FROM
				}
				add_to_war = {
					targeted_alliance = FROM
					enemy = PREV
					hostility_reason = asked_to_join
				}
			}
		}
	}
}

NSM_leave_national_front = {
	leave_faction = yes
	set_country_flag = NSM_left_faction_flag
	if = {
		limit = {
			NOT = {
				tag = ATW
			}
		}
		every_other_country = {
			limit = {
				USA_is_nationalist_tag = yes
				is_in_faction = yes
			}
			declare_war_on = {
				target = ROOT
				type = civil_war
			}
		}
	}
}

NSM_national_conference_effect = {
	custom_effect_tooltip = NSM_national_conference_tooltip
	hidden_effect = {
		if = {
			limit = {
				is_faction_leader = yes
			}
			if = {
				limit = {
					check_variable = {
						var = NSM.legit_var
						value = PTF.legit_var
						compare = greater_than
					}
					check_variable = {
						var = NSM.legit_var
						value = ATW.legit_var
						compare = greater_than
					}
				}
				NSM = {
					country_event = {
						id = nsm.23
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = PTF.legit_var
						value = NSM.legit_var
						compare = greater_than
					}
					check_variable = {
						var = PTF.legit_var
						value = ATW.legit_var
						compare = greater_than
					}
				}
				PTF = {
					country_event = {
						id = ptf.3
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = ATW.legit_var
						value = NSM.legit_var
						compare = greater_than
					}
					check_variable = {
						var = ATW.legit_var
						value = PTF.legit_var
						compare = greater_than
					}
				}
				ATW = {
					country_event = atom.13
				}
			}
			else = {
				random_list = {
					10 = {
						NSM = {
							country_event = {
								id = nsm.23
							}
						}
					}

				}
