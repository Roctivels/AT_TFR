USC_faction_var_balancer = {
	if = {
		limit = {
			check_variable = {
				target = 1
			}
		}
		divide_temp_variable = {
			USC_loyalty_lib_var_temp = 5
		}
		multiply_temp_variable = {
			USC_loyalty_lib_var_temp = -1
		}
		add_to_variable = {
			USC_loyalty_con_var_temp = USC_loyalty_lib_var_temp
		}
		add_to_variable = {
			USC_loyalty_mil_var_temp = USC_loyalty_lib_var_temp
		}
	}
	else_if = {
		limit = {
			check_variable = {
				target = 2
			}
		}
		divide_temp_variable = {
			USC_loyalty_con_var_temp = 5
		}
		multiply_temp_variable = {
			USC_loyalty_con_var_temp = -1
		}
		add_to_variable = {
			USC_loyalty_lib_var_temp = USC_loyalty_con_var_temp
		}
		add_to_variable = {
			USC_loyalty_mil_var_temp = USC_loyalty_con_var_temp
		}
	}
	else = {
		divide_temp_variable = {
			USC_loyalty_mil_var_temp = 5
		}
		multiply_temp_variable = {
			USC_loyalty_mil_var_temp = -1
		}
		add_to_variable = {
			USC_loyalty_con_var_temp = USC_loyalty_mil_var_temp
		}
		add_to_variable = {
			USC_loyalty_lib_var_temp = USC_loyalty_mil_var_temp
		}
	}
	set_temp_variable = {
		total_popularity = 0
	}
	clamp_variable = {
		var = USC_loyalty_con_var
		min = 0
		max = 1
	}
	add_to_temp_variable = {
		total_popularity = USC_loyalty_con_var
	}
	clamp_variable = {
		var = USC_loyalty_lib_var
		min = 0
		max = 1
	}
	add_to_temp_variable = {
		total_popularity = USC_loyalty_lib_var
	}
	clamp_variable = {
		var = USC_loyalty_mil_var
		min = 0
		max = 1
	}
	add_to_temp_variable = {
		total_popularity = USC_loyalty_mil_var
	}
	if = {
		limit = {
			check_variable = {
				total_popularity > 1
			}
		}
		subtract_from_temp_variable = {
			total_popularity = 1
		}
		hidden_effect = {
			random_list = {
				10 = {
					modifier = {
						factor = 0
						check_variable = {
							USC_loyalty_lib_var < total_popularity
						}
					}
					subtract_from_variable = {
						USC_loyalty_lib_var = total_popularity
					}
				}
				10 = {
					modifier = {
						factor = 0
						check_variable = {
							USC_loyalty_con_var < total_popularity
						}
					}
					subtract_from_variable = {
						USC_loyalty_con_var = total_popularity
					}
				}
				10 = {
					modifier = {
						factor = 0
						check_variable = {
							USC_loyalty_mil_var < total_popularity
						}
					}
					subtract_from_variable = {
						USC_loyalty_mil_var = total_popularity
					}
				}
			}
		}
	}
	else_if = {
		limit = {
			check_variable = {
				total_popularity < 1
			}
		}
		subtract_from_temp_variable = {
			total_popularity = 1
		}
		multiply_temp_variable = {
			total_popularity = -1
		}
		hidden_effect = {
			random_list = {
				10 = {
					add_to_variable = {
						USC_loyalty_lib_var = total_popularity
					}
				}
				10 = {
					add_to_variable = {
						USC_loyalty_mil_var = total_popularity
					}
				}
				10 = {
					add_to_variable = {
						USC_loyalty_con_var = total_popularity
					}
				}
			}
		}
	}
	set_temp_variable = {
		USC_loyalty_con_var_temp = 0
	}
	set_temp_variable = {
		USC_loyalty_lib_var_temp = 0
	}
	set_temp_variable = {
		USC_loyalty_mil_var_temp = 0
	}

		if = {
			limit = {
				has_country_flag = USC_ideology_change_enabled_flag
			}

		}
	}
