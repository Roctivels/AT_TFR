GER_decrease_corona = {
	if = {
		limit = {
			has_idea = higher_covid_cases
		}
		swap_ideas = {
			remove_idea = higher_covid_cases
			add_idea = high_covid_cases
		}
	}
	else_if = {
		limit = {
			has_idea = high_covid_cases
		}
		swap_ideas = {
			remove_idea = high_covid_cases
			add_idea = medium_covid_cases
		}
	}
	else_if = {
		limit = {
			has_idea = medium_covid_cases
		}
		swap_ideas = {
			remove_idea = medium_covid_cases
			add_idea = low_covid_cases
		}
	}
	else_if = {
		limit = {
			has_idea = low_covid_cases
		}
		swap_ideas = {
			remove_idea = low_covid_cases
			add_idea = lower_covid_cases
		}
	}
	else_if = {
		limit = {
			has_idea = lower_covid_cases
		}
		remove_ideas = lower_covid_cases
	}
}

GER_increase_militarism = {
	if = {
		limit = {
			NOT = {
				OR = {
					has_idea = GER_militarism

GER_increase_catharsis = {
	add_to_variable = {
		var = GER_mass_catharsis_society_development_monthly_dynamic_var
		value = -0.01
	}
	add_to_variable = {
		var = GER_mass_catharsis_stability_weekly_dynamic_var
		value = -0.001
	}
	add_to_variable = {
		var = GER_mass_catharsis_war_support_weekly_dynamic_var
		value = -0.001
	}
	add_to_variable = {
		var = GER_mass_catharsis_production_factory_max_efficiency_factor_dynamic_var
		value = -0.05
	}
	add_to_variable = {
		var = GER_mass_catharsis_production_factory_efficiency_gain_factor_dynamic_var
		value = -0.05
	}
	add_to_variable = {
		var = GER_mass_catharsis_resistance_growth_dynamic_var
		value = 0.05
	}
	add_to_variable = {
		var = GER_mass_catharsis_compliance_growth_dynamic_var
		value = -0.05
	}
	if = {
		limit = {
			NOT = {
				has_dynamic_modifier = {
					modifier = GER_mass_catharsis_dynamic
				}
			}
		}
		effect_tooltip = {
			add_ideas = GER_catharsis1
		}
		hidden_effect = {
			add_dynamic_modifier = {
				modifier = GER_mass_catharsis_dynamic
			}
		}
	}
	else = {
		effect_tooltip = {
			swap_ideas = {
				remove_idea = GER_catharsis_empty_dummy_idea
				add_idea = GER_catharsis1
			}
		}
	}
}
