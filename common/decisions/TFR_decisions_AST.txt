	AST_the_devastation_spreads = {
		
		icon = GFX_AST_the_devastation_spreads
		
		priority = 1

		available = {
			custom_trigger_tooltip = {
				tooltip = AST_the_devastation_spreads_tt		
				AND = {
					check_variable = { NSW_bushfire_devastation_rate = 0 }
					check_variable = { NSW_drought_devastation_rate = 0 }
					check_variable = { VIC_bushfire_devastation_rate = 0 }
					check_variable = { VIC_drought_devastation_rate = 0 }
				}
			}
		}
		
		ai_will_do = {
			factor = 200
		}

		activation = {
			has_country_flag = AST_National_Devastation_flag
		}
		
		is_good = no
		days_mission_timeout = 30
		days_re_enable = 1
		
		complete_effect = {
			set_country_flag = AST_DEVASTATION_ACTION_FLAG
		}

		timeout_effect = {
			clr_country_flag = AST_DEVASTATION_ACTION_FLAG #A truly beautiful example of a cascading error was here due to forgetting the = , amazing - Temuzu
			
			add_to_variable = {
				var = NSW_bushfire_devastation
				value = NSW_bushfire_devastation_rate
				tooltip = NSW_DS_bushfire_devastation_tt
			}
			add_to_variable = {
				var = NSW_drought_devastation
				value = NSW_drought_devastation_rate
				tooltip = NSW_DS_drought_devastation_tt
			}
			add_to_variable = {
				var = VIC_bushfire_devastation
				value = VIC_bushfire_devastation_rate
				tooltip = VIC_DS_bushfire_devastation_tt
			}
			add_to_variable = {
				var = VIC_drought_devastation
				value = VIC_drought_devastation_rate
				tooltip = VIC_DS_drought_devastation_tt
			}
			random_list = {
				15 = {
					add_to_variable = {
						var = VIC_bushfire_devastation_rate
						value = 1
						tooltip = NSW_DS_bushfire_devastation_rate_tt
					}
				}
				10 = {
					add_to_variable = {
						var = VIC_bushfire_devastation_rate
						value = 1
						tooltip = NSW_DS_drought_devastation_rate_tt
					}
				}
				10 = {
					add_to_variable = {
						var = VIC_bushfire_devastation_rate
						value = 1
						tooltip = VIC_DS_bushfire_devastation_rate_tt
					}
				}
				5 = {
					add_to_variable = {
						var = VIC_bushfire_devastation_rate
						value = 1
						tooltip = VIC_DS_drought_devastation_rate_tt
					}
				}
				60 = {
				
				}
			}
		}
