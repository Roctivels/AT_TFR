is_high_command = {
	custom_override_tooltip = {
		tooltip = is_high_command
		has_advisor_role = high_command
		is_hired_as_advisor = yes

		#make sure they're not just hired as another slot
		NOT = { not_already_hired_except_as = army_chief }
		NOT = { not_already_hired_except_as = navy_chief }
		NOT = { not_already_hired_except_as = air_chief }
		NOT = { not_already_hired_except_as = theorist }
	}
}

PRC_has_oil_crisis_trigger = {
	custom_trigger_tooltip = {
		tooltip = PRC_has_oil_crisis_trigger_tooltip
		if = {
			limit = {
				tag = PRC
			}
			OR = {
				has_idea = PRC_oil_crash1
				has_idea = PRC_oil_crash2
				has_idea = PRC_oil_crash3
				has_idea = PRC_oil_crash4
				has_idea = PRC_oil_crash5
			}
		}
		if = {
			limit = {
				tag = GER
			}
			OR = {
				has_idea = GER_oil_crash1
				has_idea = GER_oil_crash2
				has_idea = GER_oil_crash3
				has_idea = GER_oil_crash4
				has_idea = GER_oil_crash5
			}
		}
	}
