BRN_can_increase_infiltration_in_state = {
	custom_trigger_tooltip = {
		tooltip = BRN_can_be_infiltrated
		OR = {
			NOT = { has_state_flag = BRN_infiltration_@ROOT }
			has_state_flag = {
				flag = BRN_infiltration_@ROOT
				value < 4
			}
		}
	}
}

BRN_uprising_is_available = {
	custom_trigger_tooltip = {
		tooltip = BRN_uprising_is_available
		TEX = {
			any_controlled_state = {
				BRN_is_infiltrated_by_ROOT = yes
			}
		}
	}
}

BRN_is_infiltrated_by_ROOT = {
	has_state_flag = {
		flag = BRN_infiltration_@ROOT
		value > 0
	}
}

BRN_is_infiltrated_by_PREV = {
	has_state_flag = {
		flag = BRN_infiltration_@PREV
		value > 0
	}
}

BRN_is_not_infiltrating_a_state = {
	custom_trigger_tooltip = {
		tooltip = BRN_is_not_infiltrating_a_state
		NOT = {
			has_decision = BRN_infiltrate_el_paso
		}
	}
}
